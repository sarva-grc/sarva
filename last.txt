    <script>
    (function () {
        var ISL_PHOTOS = [27, 28, 30, 33, 34, 35, 38, 39, 40];
        var islIndex   = 0;

        function buildThumbs() {
            var html = '';
            for (var i = 0; i < ISL_PHOTOS.length; i++) {
                html += '<img src="images/others/' + ISL_PHOTOS[i] + '.jpg"'
                      + ' class="isl-thumb' + (i === 0 ? ' isl-active' : '') + '"'
                      + ' alt="ISL ' + ISL_PHOTOS[i] + '"'
                      + ' onclick="islSetPhoto(' + i + ')">';
            }
            document.getElementById('islThumbs').innerHTML = html;
        }

        window.islOpenAlbum = function (startIndex) {
            islIndex = startIndex || 0;
            buildThumbs();
            islSetPhoto(islIndex);
            document.getElementById('islAlbumModal').classList.add('isl-open');
            document.body.style.overflow = 'hidden';
        };

        window.islSetPhoto = function (idx) {
            islIndex = idx;
            document.getElementById('islMainImg').src =
                'images/others/' + ISL_PHOTOS[idx] + '.jpg';
            document.getElementById('islCounter').textContent =
                (idx + 1) + ' / ' + ISL_PHOTOS.length;
            var thumbs = document.querySelectorAll('.isl-thumb');
            for (var t = 0; t < thumbs.length; t++) {
                thumbs[t].classList.toggle('isl-active', t === idx);
            }
        };

        window.islNav = function (dir) {
            islSetPhoto((islIndex + dir + ISL_PHOTOS.length) % ISL_PHOTOS.length);
        };

        window.islCloseAlbum = function () {
            document.getElementById('islAlbumModal').classList.remove('isl-open');
            document.body.style.overflow = '';
        };

        window.islOutsideClose = function (e) {
            if (e.target === document.getElementById('islAlbumModal')) {
                islCloseAlbum();
            }
        };

        /* Keyboard support */
        document.addEventListener('keydown', function (e) {
            var modal = document.getElementById('islAlbumModal');
            if (!modal.classList.contains('isl-open')) return;
            if (e.key === 'ArrowRight') islNav(1);
            if (e.key === 'ArrowLeft')  islNav(-1);
            if (e.key === 'Escape')     islCloseAlbum();
        });
    })();
    </script>
